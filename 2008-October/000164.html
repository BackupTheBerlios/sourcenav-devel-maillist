you could try,,

make distclean 

./configure --prefix=/opt/snav2remove
make
make install

find /opt/snav2remove > files2remove

rm -rf /opt/snav2remove

for file in `cat files2remove`
do
  echo basename $file > filelist2remove
done

for file in `cat filelist2remove`
do
  locate $file > whereitisonmysystem
done

then verify that the files in whereitisonmysystem are ones you want to
remove, then loop through them removing them


if you don't have locate configured on your box, you can usually add it
pretty easily or you can replace the locate call with find calls...

find /bin -name  $file > whereitisonmysystem
find /lib -name  $file > whereitisonmysystem
find /usr/bin -name $file > whereitisonmysystem
find /usr/lib -name $file > whereitisonmysystem
... other appropriate top level dirs...
